\tikzset{
    cash-flow/.style = {
        line width=0.25cm,
    },
    contract/.style n args = {4}{
        draw,
        circle,
        fill={rgb:red,#2;green,#4},
        minimum width=1cm,
        minimum height=1cm,
        append after command={
            \pgfextra{
                \tikzmath{
                    \outLength=#2/10;
                    \inLength=#4/10;
                }
                \draw[cash-flow, color={rgb:red,1}]
                    (\tikzlastnode.#1) -- +(#1:\outLength);
                \draw[cash-flow, color={rgb:green,1}]
                    (\tikzlastnode.#3) -- +(#3:\inLength);    
            }
        }
    }
};
